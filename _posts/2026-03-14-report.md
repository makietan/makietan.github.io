// 最大公約数を求める関数（分数簡約用）
function gcd(a, b) {
    return b ? gcd(b, a % b) : Math.abs(a);
}

// 分数に変換する関数
function toFraction(num) {
    if (Number.isInteger(num)) return num.toString();
    const tolerance = 1.0e-9;
    let h1 = 1, h2 = 0, k1 = 0, k2 = 1;
    let b = num;
    do {
        let a = Math.floor(b);
        let aux = h1; h1 = a * h1 + h2; h2 = aux;
        aux = k1; k1 = a * k1 + k2; k2 = aux;
        b = 1 / (b - a);
    } while (Math.abs(num - h1 / k1) > num * tolerance);
    
    const commonGcd = gcd(h1, k1);
    const top = h1 / commonGcd;
    const bottom = k1 / commonGcd;
    
    if (bottom === 1) return `${top}`;
    return `\\frac{${top}}{${bottom}}`;
}

let chartInstance = null;

function updateDashboard() {
    const table = document.getElementById('dataTable');
    const rows = table.querySelectorAll('tbody tr');
    
    let xData = [];
    let eachValues = [];
    let yData = [];
    let currentSum = 0;

    // 1. テーブルから値を取得し、累計値を再計算して更新
    rows.forEach((row, index) => {
        const x = index + 1;
        const eachVal = parseFloat(row.cells[1].innerText) || 0;
        currentSum += eachVal;
        
        row.cells[0].innerText = x; // 話数固定
        row.cells[2].innerText = currentSum; // 累計値を動的書き換え
        
        xData.push(x);
        eachValues.push(eachVal);
        yData.push(currentSum);
    });

    // 2. 最小二乗法で一次近似（y = mx + c）を計算
    const n = xData.length;
    let sumX = 0, sumY = 0, sumXY = 0, sumXX = 0;
    for(let i=0; i<n; i++) {
        sumX += xData[i];
        sumY += yData[i];
        sumXY += xData[i] * yData[i];
        sumXX += xData[i] * xData[i];
    }
    
    const m = (n * sumXY - sumX * sumY) / (n * sumXX - sumX * sumX);
    const c = (sumY - m * sumX) / n;

    // 近似直線のプロットデータ作成
    let trendData = xData.map(x => m * x + c);

    // 3. KaTeXによる数式レンダリング
    const signFraction = c >= 0 ? '+' : '';
    const signDecimal = c >= 0 ? '+' : '';
    
    katex.render(`y = ${toFraction(m)}x ${signFraction} ${toFraction(c)}`, document.getElementById('math-fraction'));
    document.getElementById('math-decimal').innerHTML = `<code>y = ${m}x ${signDecimal} ${c}</code>`;

    // 4. Chart.jsでのグラフ描画・更新
    if (chartInstance) {
        chartInstance.data.labels = xData;
        chartInstance.data.datasets[0].data = yData;
        chartInstance.data.datasets[1].data = trendData;
        chartInstance.update();
    } else {
        const ctx = document.getElementById('myChart').getContext('2d');
        chartInstance = new Chart(ctx, {
            type: 'line',
            data: {
                labels: xData,
                datasets: [
                    {
                        label: '実際の累計値 (y)',
                        data: yData,
                        borderColor: '#3b82f6',
                        backgroundColor: 'rgba(59, 130, 246, 0.1)',
                        borderWidth: 2,
                        tension: 0.1,
                        fill: true
                    },
                    {
                        label: '近似直線 (トレンド)',
                        data: trendData,
                        borderColor: '#ef4444',
                        borderDash: [5, 5],
                        borderWidth: 2,
                        pointRadius: 0,
                        fill: false
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                scales: {
                    x: { title: { display: true, text: '話数 (x)' } },
                    y: { title: { display: true, text: '累積値 (y)' }, beginAtZero: true }
                }
            }
        });
    }
}

// テーブル入力が書き換わったらリアルタイムでリロードするイベントを設定
document.getElementById('dataTable').addEventListener('input', updateDashboard);

// 初回読み込み時の実行
window.onload = updateDashboard;
